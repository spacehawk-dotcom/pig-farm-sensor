// 1. 파이어베이스 실시간 데이터베이스 연결 const db = firebase.database(); const sensorRef = db.ref('sensor_logs'); // 2. 데이터가 업데이트될 때마다 자동으로 화면 갱신 sensorRef.on('value', (snapshot) => { const data = snapshot.val(); if (data) { // "창고 데이터 비어있음" 알림 숨기기 const warning = document.getElementById('empty-warning'); if (warning) warning.style.display = 'none'; // app.js에서 보낸 구역 이름들을 하나씩 순회하며 화면에 배치 Object.keys(data).forEach(zoneName => { const zone = data[zoneName]; // HTML 내의 ID가 '이유_1배치_temp' 형식이어야 합니다. const tempEl = document.getElementById(`${zoneName}_temp`); const humiEl = document.getElementById(`${zoneName}_humi`); const timeEl = document.getElementById(`${zoneName}_time`); if (tempEl) tempEl.innerText = zone.temp; if (humiEl) humiEl.innerText = zone.humi; if (timeEl) { // '연결실패'인 경우와 정상인 경우를 구분해서 표시 timeEl.innerText = zone.timestamp; timeEl.style.color = (zone.timestamp === "연결실패") ? "red" : "#888"; } }); } }, (error) => { console.error("데이터 읽기 실패:", error); });
